2. BFS — Breadth-First Search

Pattern: Level-order traversal / Shortest path in unweighted graphs

pythonfrom collections import deque

def bfs(graph, start):
    visited = {start}
    queue = deque([start])
    while queue:
        node = queue.popleft()
        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)

BFS with Distance Tracking

pythondef bfs_distance(graph, start):
    dist = {start: 0}
    queue = deque([start])
    while queue:
        node = queue.popleft()
        for neighbor in graph[node]:
            if neighbor not in dist:
                dist[neighbor] = dist[node] + 1
                queue.append(neighbor)
    return dist

Multi-Source BFS

python# Start BFS from multiple sources simultaneously
# Used for: "minimum distance from any source" problems
def multi_source_bfs(grid, sources):
    queue = deque()
    dist = {}
    for r, c in sources:
        queue.append((r, c))
        dist[(r, c)] = 0
    while queue:
        r, c = queue.popleft()
        for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < len(grid) and 0 <= nc < len(grid[0]) and (nr, nc) not in dist:
                dist[(nr, nc)] = dist[(r, c)] + 1
                queue.append((nr, nc))
    return dist

0-1 BFS (Deque-based, weights 0 or 1)

pythondef bfs_01(graph, start, n):
    dist = [float('inf')] * n
    dist[start] = 0
    dq = deque([start])
    while dq:
        u = dq.popleft()
        for v, w in graph[u]:  # w is 0 or 1
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                if w == 0:
                    dq.appendleft(v)
                else:
                    dq.append(v)
    return dist

3. DFS — Depth-First Search

Pattern: Explore as deep as possible, backtrack

python# Recursive DFS
def dfs(graph, node, visited):
    visited.add(node)
    for neighbor in graph[node]:
        if neighbor not in visited:
            dfs(graph, neighbor, visited)

# Iterative DFS
def dfs_iterative(graph, start):
    visited = set()
    stack = [start]
    while stack:
        node = stack.pop()
        if node in visited:
            continue
        visited.add(node)
        for neighbor in graph[node]:
            if neighbor not in visited:
                stack.append(neighbor)

DFS on Grid

pythondef dfs_grid(grid, r, c, visited):
    if r < 0 or r >= len(grid) or c < 0 or c >= len(grid[0]):
        return
    if (r, c) in visited or grid[r][c] == 0:
        return
    visited.add((r, c))
    for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
        dfs_grid(grid, r+dr, c+dc, visited)

4. Cycle Detection

Undirected Graph — DFS

pythondef has_cycle_undirected(graph, n):
    visited = set()

    def dfs(node, parent):
        visited.add(node)
        for neighbor in graph[node]:
            if neighbor not in visited:
                if dfs(neighbor, node):
                    return True
            elif neighbor != parent:
                return True  # back edge = cycle
        return False

    for i in range(n):
        if i not in visited:
            if dfs(i, -1):
                return True
    return False

Directed Graph — DFS with 3 Colors

python# WHITE=0 (unvisited), GRAY=1 (in current path), BLACK=2 (fully processed)
def has_cycle_directed(graph, n):
    color = [0] * n

    def dfs(u):
        color[u] = 1  # GRAY
        for v in graph[u]:
            if color[v] == 1:
                return True    # back edge → cycle
            if color[v] == 0 and dfs(v):
                return True
        color[u] = 2  # BLACK
        return False

    return any(color[i] == 0 and dfs(i) for i in range(n))

Directed Graph — Kahn's (BFS Topological Sort)

python# If topological sort doesn't include all nodes → cycle exists
def has_cycle_kahn(graph, n):
    indegree = [0] * n
    for u in range(n):
        for v in graph[u]:
            indegree[v] += 1
    queue = deque(i for i in range(n) if indegree[i] == 0)
    count = 0
    while queue:
        u = queue.popleft()
        count += 1
        for v in graph[u]:
            indegree[v] -= 1
            if indegree[v] == 0:
                queue.append(v)
    return count != n  # True if cycle

5. Topological Sort

Pattern: Linear ordering of vertices in a DAG such that for every edge u→v, u comes before v.

Kahn's Algorithm (BFS — preferred for interviews)

pythonfrom collections import deque

def topo_sort_kahn(graph, n):
    indegree = [0] * n
    for u in range(n):
        for v in graph[u]:
            indegree[v] += 1

    queue = deque(i for i in range(n) if indegree[i] == 0)
    order = []

    while queue:
        u = queue.popleft()
        order.append(u)
        for v in graph[u]:
            indegree[v] -= 1
            if indegree[v] == 0:
                queue.append(v)

    return order if len(order) == n else []  # empty if cycle

DFS-based Topological Sort

pythondef topo_sort_dfs(graph, n):
    visited = set()
    stack = []

    def dfs(u):
        visited.add(u)
        for v in graph[u]:
            if v not in visited:
                dfs(v)
        stack.append(u)  # post-order

    for i in range(n):
        if i not in visited:
            dfs(i)

    return stack[::-1]  # reverse post-order

6. Shortest Path Algorithms

6a. Dijkstra's Algorithm

When: Non-negative weights, single source. Complexity: O((V + E) log V) with min-heap.
pythonimport heapq

def dijkstra(graph, start, n):
    dist = [float('inf')] * n
    dist[start] = 0
    heap = [(0, start)]  # (distance, node)

    while heap:
        d, u = heapq.heappop(heap)
        if d > dist[u]:
            continue  # skip outdated entries
        for v, w in graph[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                heapq.heappush(heap, (dist[v], v))

    return dist

6b. Bellman-Ford

When: Negative weights allowed, single source. Detects negative cycles. Complexity: O(V × E)
pythondef bellman_ford(n, edges, start):
    dist = [float('inf')] * n
    dist[start] = 0

    for _ in range(n - 1):
        for u, v, w in edges:
            if dist[u] != float('inf') and dist[u] + w < dist[v]:
                dist[v] = dist[u] + w

    # Negative cycle detection
    for u, v, w in edges:
        if dist[u] != float('inf') and dist[u] + w < dist[v]:
            return None  # negative cycle exists

    return dist

6c. Floyd-Warshall

When: All-pairs shortest path. Handles negative weights (no negative cycles). Complexity: O(V³)
pythondef floyd_warshall(n, edges):
    dist = [[float('inf')] * n for _ in range(n)]
    for i in range(n):
        dist[i][i] = 0
    for u, v, w in edges:
        dist[u][v] = w

    for k in range(n):
        for i in range(n):
            for j in range(n):
                if dist[i][k] + dist[k][j] < dist[i][j]:
                    dist[i][j] = dist[i][k] + dist[k][j]

    return dist

6d. SPFA (Shortest Path Faster Algorithm)

When: Bellman-Ford optimization with queue. Average O(E), worst O(VE).
pythondef spfa(graph, start, n):
    dist = [float('inf')] * n
    dist[start] = 0
    in_queue = [False] * n
    queue = deque([start])
    in_queue[start] = True

    while queue:
        u = queue.popleft()
        in_queue[u] = False
        for v, w in graph[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                if not in_queue[v]:
                    queue.append(v)
                    in_queue[v] = True
    return dist

6e. Shortest Path in DAG (Topo Sort + Relaxation)

pythondef shortest_path_dag(graph, n, start):
    order = topo_sort_kahn(graph, n)
    dist = [float('inf')] * n
    dist[start] = 0
    for u in order:
        if dist[u] != float('inf'):
            for v, w in graph[u]:
                dist[v] = min(dist[v], dist[u] + w)
    return dist

7. Minimum Spanning Tree

7a. Kruskal's Algorithm (Edge-based, uses Union-Find)

Complexity: O(E log E)
pythondef kruskal(n, edges):
    edges.sort(key=lambda x: x[2])  # sort by weight
    parent = list(range(n))
    rank = [0] * n

    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]  # path compression
            x = parent[x]
        return x

    def union(a, b):
        ra, rb = find(a), find(b)
        if ra == rb:
            return False
        if rank[ra] < rank[rb]:
            ra, rb = rb, ra
        parent[rb] = ra
        if rank[ra] == rank[rb]:
            rank[ra] += 1
        return True

    mst_cost = 0
    mst_edges = 0
    for u, v, w in edges:
        if union(u, v):
            mst_cost += w
            mst_edges += 1
            if mst_edges == n - 1:
                break
    return mst_cost

7b. Prim's Algorithm (Vertex-based, uses min-heap)

Complexity: O((V + E) log V)
pythondef prim(graph, n):
    visited = [False] * n
    heap = [(0, 0)]  # (weight, node), start from node 0
    mst_cost = 0

    while heap:
        w, u = heapq.heappop(heap)
        if visited[u]:
            continue
        visited[u] = True
        mst_cost += w
        for v, weight in graph[u]:
            if not visited[v]:
                heapq.heappush(heap, (weight, v))

    return mst_cost

When to Use

Kruskal's

Best for sparse graphs and when the edge list is given.

Prim's

Best for dense graphs and when the adjacency list is given.

8. Union-Find / Disjoint Set Union

Template (Path Compression + Union by Rank)

pythonclass UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n
        self.components = n

    def find(self, x):
        while self.parent[x] != x:
            self.parent[x] = self.parent[self.parent[x]]  # path halving
            x = self.parent[x]
        return x

    def union(self, a, b):
        ra, rb = self.find(a), self.find(b)
        if ra == rb:
            return False
        if self.rank[ra] < self.rank[rb]:
            ra, rb = rb, ra
        self.parent[rb] = ra
        if self.rank[ra] == self.rank[rb]:
            self.rank[ra] += 1
        self.components -= 1
        return True

    def connected(self, a, b):
        return self.find(a) == self.find(b)

Weighted Union-Find

pythonclass WeightedUF:
    def __init__(self, n):
        self.parent = list(range(n))
        self.weight = [0.0] * n  # weight[x] = ratio from x to parent[x]

    def find(self, x):
        if self.parent[x] != x:
            root = self.find(self.parent[x])
            self.weight[x] *= self.weight[self.parent[x]]
            self.parent[x] = root
        return self.parent[x]

    def union(self, a, b, ratio):  # a / b = ratio
        ra, rb = self.find(a), self.find(b)
        if ra == rb:
            return
        self.parent[ra] = rb
        self.weight[ra] = ratio * self.weight[b] / self.weight[a]

9. Bipartite Graph

Pattern: Can the graph be 2-colored? (No odd-length cycles)

BFS Coloring

pythondef is_bipartite(graph, n):
    color = [-1] * n
    for start in range(n):
        if color[start] != -1:
            continue
        queue = deque([start])
        color[start] = 0
        while queue:
            u = queue.popleft()
            for v in graph[u]:
                if color[v] == -1:
                    color[v] = 1 - color[u]
                    queue.append(v)
                elif color[v] == color[u]:
                    return False
    return True

10. Strongly Connected Components

Kosaraju's Algorithm

Strategy: Find all SCCs in a directed graph. Two DFS passes.
pythondef kosaraju(graph, reverse_graph, n):
    # Pass 1: Get finish order
    visited = set()
    order = []

    def dfs1(u):
        visited.add(u)
        for v in graph[u]:
            if v not in visited:
                dfs1(v)
        order.append(u)

    for i in range(n):
        if i not in visited:
            dfs1(i)

    # Pass 2: Process in reverse finish order on transposed graph
    visited.clear()
    sccs = []

    def dfs2(u, component):
        visited.add(u)
        component.append(u)
        for v in reverse_graph[u]:
            if v not in visited:
                dfs2(v, component)

    for u in reversed(order):
        if u not in visited:
            component = []
            dfs2(u, component)
            sccs.append(component)

    return sccs

Tarjan's Algorithm

Strategy: Single DFS pass, uses low-link values.
pythondef tarjan_scc(graph, n):
    idx = [0]
    stack = []
    on_stack = [False] * n
    ids = [-1] * n
    low = [0] * n
    sccs = []

    def dfs(u):
        ids[u] = low[u] = idx[0]
        idx[0] += 1
        stack.append(u)
        on_stack[u] = True

        for v in graph[u]:
            if ids[v] == -1:
                dfs(v)
                low[u] = min(low[u], low[v])
            elif on_stack[v]:
                low[u] = min(low[u], ids[v])

        # Root of SCC
        if low[u] == ids[u]:
            component = []
            while True:
                v = stack.pop()
                on_stack[v] = False
                component.append(v)
                if v == u:
                    break
            sccs.append(component)

    for i in range(n):
        if ids[i] == -1:
            dfs(i)

    return sccs

11. Bridges & Articulation Points

Bridges (Critical Edges)

pythondef find_bridges(graph, n):
    timer = [0]
    tin = [-1] * n
    low = [-1] * n
    bridges = []

    def dfs(u, parent):
        tin[u] = low[u] = timer[0]
        timer[0] += 1
        for v in graph[u]:
            if v == parent:
                continue
            if tin[v] == -1:
                dfs(v, u)
                low[u] = min(low[u], low[v])
                if low[v] > tin[u]:
                    bridges.append((u, v))
            else:
                low[u] = min(low[u], tin[v])

    for i in range(n):
        if tin[i] == -1:
            dfs(i, -1)

    return bridges

Articulation Points (Cut Vertices)

pythondef find_articulation_points(graph, n):
    timer = [0]
    tin = [-1] * n
    low = [-1] * n
    ap = set()

    def dfs(u, parent):
        tin[u] = low[u] = timer[0]
        timer[0] += 1
        children = 0
        for v in graph[u]:
            if v == parent:
                continue
            if tin[v] == -1:
                children += 1
                dfs(v, u)
                low[u] = min(low[u], low[v])
                if parent != -1 and low[v] >= tin[u]:
                    ap.add(u)
            else:
                low[u] = min(low[u], tin[v])
        if parent == -1 and children > 1:
            ap.add(u)

    for i in range(n):
        if tin[i] == -1:
            dfs(i, -1)

    return ap

Hierholzer's Algorithm

pythondef euler_path(graph, start):
    # graph is dict of defaultdict(list) or adjacency with mutable edges
    stack = [start]
    path = []
    while stack:
        u = stack[-1]
        if graph[u]:
            v = graph[u].pop()
            stack.append(v)
        else:
            path.append(stack.pop())
    return path[::-1]

13. Network Flow / Max Flow

Ford-Fulkerson (Edmonds-Karp variant, BFS-based)

Complexity: O(VE²)
pythondef max_flow_bfs(graph, source, sink, n):
    """Edmonds-Karp: BFS-based Ford-Fulkerson. O(VE²)"""
    # graph is capacity matrix graph[u][v] = capacity

    def bfs(source, sink, parent):
        visited = {source}
        queue = deque([source])
        while queue:
            u = queue.popleft()
            for v in range(n):
                if v not in visited and graph[u][v] > 0:
                    visited.add(v)
                    parent[v] = u
                    if v == sink:
                        return True
                    queue.append(v)
        return False

    total_flow = 0
    parent = [-1] * n

    while bfs(source, sink, parent):
        # Find bottleneck
        path_flow = float('inf')
        v = sink
        while v != source:
            u = parent[v]
            path_flow = min(path_flow, graph[u][v])
            v = u

        # Update residual graph
        v = sink
        while v != source:
            u = parent[v]
            graph[u][v] -= path_flow
            graph[v][u] += path_flow
            v = u

        total_flow += path_flow
        parent = [-1] * n

    return total_flow

  • Bipartite Matching (maximum matching = max flow in bipartite network)

14. Graph Coloring

M-Coloring (Backtracking)

pythondef can_color(graph, n, m):
    color = [0] * n

    def is_safe(node, c):
        return all(color[v] != c for v in graph[node])

    def solve(node):
        if node == n:
            return True
        for c in range(1, m + 1):
            if is_safe(node, c):
                color[node] = c
                if solve(node + 1):
                    return True
                color[node] = 0
        return False

    return solve(0)

15a. BFS/DFS with State (Bitmask)

python# Shortest path visiting all nodes — BFS with (node, visited_bitmask) state
def shortest_path_all_nodes(graph, n):
    full_mask = (1 << n) - 1
    queue = deque()
    visited = set()

    for i in range(n):
        state = (i, 1 << i)
        queue.append((i, 1 << i, 0))  # node, mask, dist
        visited.add(state)

    while queue:
        node, mask, dist = queue.popleft()
        if mask == full_mask:
            return dist
        for nei in graph[node]:
            new_mask = mask | (1 << nei)
            if (nei, new_mask) not in visited:
                visited.add((nei, new_mask))
                queue.append((nei, new_mask, dist + 1))
    return -1

15c. Bidirectional BFS

pythondef bidirectional_bfs(graph, start, end):
    if start == end:
        return 0
    front = {start}
    back = {end}
    visited = {start, end}
    steps = 0

    while front and back:
        steps += 1
        # Always expand the smaller frontier
        if len(front) > len(back):
            front, back = back, front
        next_front = set()
        for node in front:
            for nei in graph[node]:
                if nei in back:
                    return steps
                if nei not in visited:
                    visited.add(nei)
                    next_front.add(nei)
        front = next_front

    return -1  # no path

15g. LCA — Lowest Common Ancestor

python# Binary Lifting: O(n log n) preprocess, O(log n) per query
import math

def preprocess_lca(graph, root, n):
    LOG = int(math.log2(n)) + 1
    depth = [0] * n
    up = [[0] * n for _ in range(LOG)]

    def dfs(v, p, d):
        depth[v] = d
        up[0][v] = p
        for k in range(1, LOG):
            up[k][v] = up[k-1][up[k-1][v]]
        for u in graph[v]:
            if u != p:
                dfs(u, v, d + 1)

    dfs(root, root, 0)

    def lca(a, b):
        if depth[a] < depth[b]:
            a, b = b, a
        diff = depth[a] - depth[b]
        for k in range(LOG):
            if (diff >> k) & 1:
                a = up[k][a]
        if a == b:
            return a
        for k in range(LOG - 1, -1, -1):
            if up[k][a] != up[k][b]:
                a = up[k][a]
                b = up[k][b]
        return up[0][a]

    return lca
